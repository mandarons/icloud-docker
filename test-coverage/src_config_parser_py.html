25def traverse_config_path(config, config_path: list[str]) -> bool: 

26 """Traverse given config path.""" 

27 if len(config_path) == 0: 

28 return True 

29 if not (config and config_path[0] in config): 

30 return False 

31 return traverse_config_path(config[config_path[0]], config_path=config_path[1:])

41def get_username(config): 

42 """Get usename from config."""

45 if not traverse_config_path(config=config, config_path=config_path):

285def get_photos_filters(config): 

286 """Return photos filters from config.""" 

287 photos_filters = { 

288 "libraries": None, 

289 "albums": None, 

290 "file_sizes": ["original"], 

291 "extensions": None, 

292 } 

293 valid_file_sizes = list(PhotoAsset.PHOTO_VERSION_LOOKUP.keys()) 

294 config_path = ["photos", "filters"] 

295 

296 # Check for filters 

297 if not traverse_config_path(config=config, config_path=config_path): 

298 LOGGER.warning( 

299 f"{config_path_to_string(config_path=config_path)} not found. \ 

300 Downloading all libraries and albums with original size ..." 

301 ) 

302 return photos_filters 

303 

304 # Parse libraries 

305 config_path.append("libraries") 

306 if ( 

307 not traverse_config_path(config=config, config_path=config_path) 

308 or not get_config_value(config=config, config_path=config_path) 

309 or len(get_config_value(config=config, config_path=config_path)) == 0 

310 ): 

311 LOGGER.warning( 

312 f"{config_path_to_string(config_path=config_path)} not found. Downloading all libraries ..." 

313 ) 

314 else: 

315 photos_filters["libraries"] = get_config_value( 

316 config=config, config_path=config_path 

317 ) 

318 

319 # Parse albums 

320 config_path[2] = "albums" 

321 if ( 

322 not traverse_config_path(config=config, config_path=config_path) 

323 or not get_config_value(config=config, config_path=config_path) 

324 or len(get_config_value(config=config, config_path=config_path)) == 0 

325 ): 

326 LOGGER.warning( 

327 f"{config_path_to_string(config_path=config_path)} not found. Downloading all albums ..." 

328 ) 

329 else: 

330 photos_filters["albums"] = get_config_value( 

331 config=config, config_path=config_path 

332 ) 

333 

334 # Parse file sizes 

335 config_path[2] = "file_sizes" 

336 if not traverse_config_path(config=config, config_path=config_path): 

337 LOGGER.warning( 

338 f"{config_path_to_string(config_path=config_path)} not found. Downloading original size photos ..." 

339 ) 

340 else: 

341 file_sizes = get_config_value(config=config, config_path=config_path) 

342 for file_size in file_sizes: 

343 if file_size not in valid_file_sizes: 

344 LOGGER.warning( 

345 f"Skipping the invalid file size {file_size}, " 

346 + f"valid file sizes are {','.join(valid_file_sizes)}." 

347 ) 

348 file_sizes.remove(file_size) 

349 if len(file_sizes) == 0: 

350 file_sizes = ["original"] 

351 photos_filters["file_sizes"] = file_sizes 

352 

353 # Parse extensions 

354 config_path[2] = "extensions" 

355 if ( 

356 not traverse_config_path(config=config, config_path=config_path) 

357 or not get_config_value(config=config, config_path=config_path) 

358 or len(get_config_value(config=config, config_path=config_path)) == 0 

359 ): 

360 LOGGER.warning( 

361 f"{config_path_to_string(config_path=config_path)} not found. Downloading all extensions ..." 

362 ) 

363 else: 

364 photos_filters["extensions"] = get_config_value( 

365 config=config, config_path=config_path 

366 ) 

367 

368 return photos_filters
